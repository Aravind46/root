function InstallDB() {

    db.transaction(function (tx) {
        tx.executeSql("CREATE TABLE IF NOT EXISTS USER_INFO(	UID int ,	USERNAME varchar(25) NULL,	MOBILE int NULL,	LONGITUDE DECIMAL,LATTITUDE DECIMAL)");
        tx.executeSql("CREATE TABLE IF NOT EXISTS TICKET(UID INT,SOURCE INT,DESTINATION INT,TIME VARCHAR(100),VALID BIGINT)");
        tx.executeSql("CREATE TABLE IF NOT EXISTS PASSES(UID INT,PASSTYPE BIGINT,TIME VARCHAR(100),VALID BIGINT)");
        tx.executeSql("CREATE TABLE IF NOT EXISTS WALLET(UID INT,BALANCE DECIMAL)");
        tx.executeSql("CREATE TABLE IF NOT EXISTS TRANSACTIONS(UID INT,TRANTYPE BIGINT,TIME VARCHAR(100))");
        tx.executeSql("CREATE TABLE IF NOT EXISTS USERTRACE(UID INT,LATTITUDE DECIMAL,LONGITUDE DECIMAL ,TIME TIMESTAMP) ");
        tx.executeSql("CREATE TABLE IF NOT EXISTS STAGEINDEX(SID INT ,NAME VARCHAR(100))");
        tx.executeSql("CREATE TABLE IF NOT EXISTS BUSINDEX(BID INT IDENTITY(1,1),BUS_NUM VARCHAR(20),SOURCE INT ,DESTINATION INT,STYPE INT,DURATION INT)");
        tx.executeSql("CREATE TABLE IF NOT EXISTS BUSROUTEINFO(BUS_NUM VARCHAR(20),STAGE INT ,STAGEORDER INT)");
    
    });
}
